    ### 3. Try Different Languages
    
    We have SDKs for JavaScript, React, and Vue. For other languages, use standard HTTP requests.
  </Col>
  <Col>
    <LanguageTabs>
      <Tab label="JavaScript">
        <CodePlayground
          language="javascript"
          code={`import { MuminClient } from '@mumin/core'

    const client = new MuminClient('YOUR_API_KEY')

    // Get a hadith cleanly with types!
    const hadith = await client.hadiths.get(1)
    console.log(hadith.translation.text)`}
          runnable
        />

        <Callout type="warning">
          **ES Modules required:** Our SDK uses modern `import` syntax. If you get an error *'Cannot use import statement'*, make sure to add `"type": "module"` to your `package.json` or rename your file to `.mjs`.
        </Callout>
      </Tab>

      <Tab label="Python">
        ```python
        import requests

        headers = {'Authorization': 'Bearer YOUR_API_KEY'}
        response = requests.get(
            'https://api.mumin.ink/v1/hadiths/1',
            params={'lang': 'en'},
            headers=headers
        )
        hadith = response.json()
        print(hadith['translations']['en']['text'])
        ```
      </Tab>

      <Tab label="PHP">
        ```php
        <?php
        $ch = curl_init();
        curl_setopt($ch, CURLOPT_URL, 'https://api.mumin.ink/v1/hadiths/1?lang=en');
        curl_setopt($ch, CURLOPT_HTTPHEADER, [
            'Authorization: Bearer YOUR_API_KEY'
        ]);
        curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);

        $response = curl_exec($ch);
        $hadith = json_decode($response);

        echo $hadith->translations->en->text;
        ?>
        ```
      </Tab>

      <Tab label="cURL">
        ```bash
        curl -X GET "https://api.mumin.ink/v1/hadiths/1?lang=en" \
          -H "Authorization: Bearer YOUR_API_KEY"
        ```
      </Tab>
    </LanguageTabs>

  </Col>
</Row>
